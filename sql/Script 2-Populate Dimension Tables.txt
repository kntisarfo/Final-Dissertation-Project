INSERT INTO dim_period (period_code, year, month)
SELECT
    TO_CHAR(d, 'YYYY') || 'M' || TO_CHAR(d, 'MM'),
    EXTRACT(YEAR FROM d)::INT,
    EXTRACT(MONTH FROM d)::INT
FROM generate_series(
    '2015-01-01'::DATE,
    '2024-12-01'::DATE,
    '1 month'::INTERVAL
) AS d;
